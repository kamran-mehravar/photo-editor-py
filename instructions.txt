from flask import Flask

def create_app():
    app = Flask(__name__)
    app.config.from_object('config')

    from app.routes import main
    app.register_blueprint(main)

    return app
۳. ساخت app/routes.py

from flask import Blueprint, render_template, request, jsonify
from app.image_processor import process_image

main = Blueprint('main', __name__)

@main.route('/')
def index():
    return render_template('index.html')

@main.route('/edit', methods=['POST'])
def edit_image():
    image = request.files['image']
    params = request.form.to_dict()
    output_path = process_image(image, params)
    return jsonify({'output': output_path})

from gimpfu import *

def my_custom_filter(image, drawable):
    """افزایش روشنایی تصویر"""
    pdb.gimp_brightness_contrast(drawable, 30, 10)
    pdb.gimp_displays_flush()

register(

    "افزایش روشنایی",

📌 گام ۷: اتصال GIMP به Flask
حالا باید Flask با GIMP ارتباط برقرار کند.
این کار را در app/image_processor.py انجام می‌دهیم:

python
Copy
Edit
import os
from PIL import Image
from subprocess import call

def process_image(image, params):
    input_path = "static/uploads/input.png"
    output_path = "static/uploads/output.png"

    image.save(input_path)

    # اجرای پلاگین GIMP برای پردازش تصویر
    call(["flatpak", "run", "org.gimp.GIMP", "--batch-interpreter=python-fu-eval", "-b",
          "import gimpfu; img = pdb.file_png_load('{}', '{}'); layer = img.active_layer; pdb.python_fu_my_custom_filter(img, layer); pdb.file_png_save(img, layer, '{}', '{}', False, 9, True, False, False, False, False);".format(input_path, input_path, output_path, output_path)])

    return output_path
✅ الان Flask می‌تواند از GIMP برای پردازش تصویر استفاده کند.
